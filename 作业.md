# 编写一个子程序，它以二进制形式显示EAX中32位数据，并设计一个主程序验证。

正确答案：
.text

.global main

main:

mov eax， -100

call dispeax

ret



dispeax：

mov ecx,32 #移位次数

mov edx, eax #保存eax

mov al, 0x30 #初始化‘0’

again: 

shl eax, 1 #从高位到低位显示

jnc disp #=0直接显示

inc al #=1 al=‘1’

disp:

call dispc #显示al中的值

mov al, 0x30 #恢复al = ‘0’

loop again #循环

ret #子程序结束

# 4. (简答题)
判断下面语句是否有错，如果有请说明错误原因。

1 mov dvar[rip+4],dvar[rip]

2 mov edi,si

3 mov eax,0x50a

4 mov al,0x50a

5 mov [rbx+4], 200 

6 mov ds, 100

7 lea eax，[rbx]

8 mov dil, ah  

9 mov edx, [ebx+0xff]

10 xchg 100, al

## 答案

正确答案：
1  错误，两个操作数不能都是内存寻址方式

2 错误，两个操作数类型不一致

3 正确

4 错误，立即数超出寄存器位长

6 错误，立即数不能给段寄存器直接赋值

7 正确

8  错误，不能同时使用原来的高字节寄存器和新增字节寄存器

10 错误，xchg指令的操作数不能是立即数

# 6. (简答题)
一段C语言程序用GCC 产生的汇编代码如下：

sub rsi, rdx

imul rdi, rsi

mov rax, rsi

sal rax, 63

sar rax, 63

xor rax,rdi

其中rdi,rsi,和rdx中分别保存变量long long变量x，y，z，写出等价于上述汇编代码的C语言代码，说明理由。



## 答案

正确答案：
前三行汇编指令的功能是：rdi=x(y-z), rax=y-z, 接下来两条移位指令的作用是用rax的最低位填充整个寄存器，如果rax最低位为0，最后一条指令保留原rdi的值，如果rax最低位为1，原rdi的值各位求反。对应的C语句如下：

(x*(y-z)) ^(0- (y-z)&1 )

或者：

long long  t1=y-z;

long  long t2=x * t1;

long  long t3=(t1<<63)>>63; 

long  long t4=t3ˆt2;
